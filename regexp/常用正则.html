空白字符和/[^A-z0-9_]+/替换为空 #

html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Document</title>
</head>
<body>
  <input type="text" id="test" onkeyup="test(this)">
  <script type="text/javascript">
    var oTest = document.getElementById('test');
		//0-9A-z_ => \w   希腊字母  俄文字母 ....
    function test(obj) {
      var reg = /\s/;         // /\D/

      obj.value = obj.value.replace(reg, '');
    }

    function test(obj) {
      var reg = /[^A-z0-9_]+/;

      obj.value = obj.value.replace(reg, '');
    }
  </script>
</body>
</html>

密码强度 #

html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Document</title>
</head>
<body>
  <input type="text" id="test">
  <button onclick="test()">点击</button>
  <script type="text/javascript">
    var oTest = document.getElementById('test');
    // 密码至少6位，包含1个大写  1个小写 1个数字 1个特殊字符
    // ?= 正向预查  正向断言
    function test() {
      var val = oTest.value,
          reg = /^.*(?=.{6,})(?=.*\d)(?=.*[A-Z])(?=.*[a-z])(?=.*[~!@#$%^&*?]).*$/,
          pass = reg.test(val);

      console.log(pass);
    }
  </script>
</body>
</html>

捕获分组 #

白话文：捕获子表达式

javascript
var str = 'abcabc',
    reg = /(b)(c)/;

    // reg = /(?:a)(b)(c)/;  // 加上 "?"和":" 不让它捕获该子表达式
    // 使用reg.exec(str)时，可以使用该方法进行剔除子表达式
console.log(str.match(reg));
// ["bc", "b", "c", index: 1, input: "abcabc", groups: undefined]
